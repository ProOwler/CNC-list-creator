	</ReportItemList>

Шаблон отчёта
// массив объектов, в объекте массив проектов, в проекте массив раскроев
// на каждом уровне статус и дата выполнения
// статусы: ожидает / готов / иное
	[{
		"item_name" : string,
		"status" : string,
		"date_ready" : date,
		"level" : int,
		"inner_items" : [{ "item_name", "status", "date_ready", "level", "inner_items" }, ... ]
	}, ... ]
